$ echo 'source /opt/homebrew/share/powerlevel10k/powerlevel10k.zsh-theme' >> ~/.zshrc
# Khởi động lại iTerm2 → wizard sẽ hỏi cài font và cấu hình prompt

Bước 2 — Cấu hình ~/.zshrc
~/.zshrc — Cấu hình tối ưu cho Backend DevConfig
# ~/.zshrc — Backend Developer config

# ── Oh My Zsh ──
ZSH="$HOME/.oh-my-zsh"
ZSH_THEME=""  # dùng Powerlevel10k thay
plugins=(
    git                      # git shortcuts: gst, gco, gp...
    zsh-autosuggestions      # gợi ý từ history
    zsh-syntax-highlighting  # highlight khi gõ
    docker                   # docker autocomplete
    docker-compose
    kubectl
    gradle
    mvn
    history                  # h alias cho history
    z                        # jump đến thư mục hay dùng
    fzf
)
source $ZSH/oh-my-zsh.sh

# ── PATH ──
export PATH="/opt/homebrew/bin:$HOME/.local/bin:$HOME/go/bin:$PATH"
export JAVA_HOME=$(/usr/libexec/java_home -v 21)

# ── fzf — giao diện tìm kiếm đẹp hơn ──
export FZF_DEFAULT_COMMAND='fd --type f --hidden --follow --exclude .git'
export FZF_DEFAULT_OPTS="--height 40% --layout=reverse --border --preview 'bat --color=always {}'"

# ── Thay thế lệnh cũ bằng tools mới ──
alias ls='eza --icons --group-directories-first'
alias ll='eza -la --icons --git --group-directories-first'
alias lt='eza --tree --level=2 --icons'
alias cat='bat --style=numbers,changes'
alias grep='rg'
alias find='fd'

# ── Git shortcuts ──
alias gs='git status'
alias gl='git log --oneline --graph --decorate --all'
alias gd='git diff'
alias gds='git diff --staged'

# ── Docker shortcuts ──
alias dk='docker'
alias dkps='docker ps --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"'
alias dkl='docker logs -f'
alias dc='docker compose'

# ── Project shortcuts ──
alias work='cd ~/Projects'
alias desk='cd ~/Desktop'
alias zshrc='code ~/.zshrc'   # mở zshrc trong VSCode
alias reload='source ~/.zshrc'  # reload config

Navigation commandsNavigation
# ── Cơ bản ──
$ pwd                          # in đường dẫn hiện tại
$ cd ~/Projects/my-service      # vào thư mục
$ cd -                          # quay lại thư mục trước đó (toggle)

# ── pushd/popd — stack thư mục ──
$ pushd /etc                   # vào /etc, lưu vị trí hiện tại vào stack
$ pushd /var/log               # vào /var/log, stack = [/etc, ~/Projects]
$ dirs -v                       # xem stack
$ popd                          # quay về /etc
$ popd                          # quay về ~/Projects

# ── z — jump thông minh (học từ history) ──
# Sau khi dùng ~/Projects/my-service vài lần:
$ z my-service                 # tự jump đến ~/Projects/my-service
$ z proj                        # jump đến thư mục gần nhất khớp với "proj"
$ zi                            # interactive mode với fzf

# ── ls với eza ──
$ ll                            # list chi tiết + git status
$ ll --sort=modified            # sắp theo ngày sửa
$ lt                            # tree view 2 cấp
$ eza --tree --level=3 src/    # tree 3 cấp của src/

# ── open trong Finder hoặc VSCode ──
$ open .                        # mở thư mục hiện tại trong Finder
$ open -a Finder ~/Projects   # mở path cụ thể
$ code .                        # mở VSCode tại thư mục hiện tại
$ idea .                        # mở IntelliJ IDEA

$ cat application.yml                 # in ra (bat nếu đã alias)
$ less large-log.txt                  # xem cuộn (q để thoát)
$ head -20 file.txt                  # 20 dòng đầu
$ tail -50 file.txt                  # 50 dòng cuối
$ tail -f app.log                    # follow log real-time
$ wc -l file.txt                     # đếm số dòng

# ── Permissions ──
$ chmod +x deploy.sh                 # thêm quyền execute
$ chmod 644 config.yml               # rw-r--r--
$ chmod 755 scripts/                 # rwxr-xr-x (thư mục scripts)
$ chown user:group file.txt         # đổi owner
$ ls -la                              # xem permissions

# ── Disk usage ──

SEARCH

Tìm kiếm & Grep — rg nhanh hơn grep 10x

ripgrep, fd, find — tìm file và nội dung trong codebase lớn

ripgrep (rg) — tìm trong code
ripgrep + fd + fzf — search power comboSearch
# ── ripgrep (rg) — tìm nội dung trong file ──
$ rg "OrderService"               # tìm trong toàn project
$ rg -i "todo"                    # case-insensitive
$ rg -t java "@Transactional"   # chỉ trong file .java
$ rg -l "TODO"                    # -l chỉ in tên file
$ rg -n "NullPointer"             # -n hiện số dòng
$ rg "password" --type-not yaml  # loại trừ yaml
$ rg -A 3 -B 3 "catch.*Exception" # 3 dòng context trước/sau
$ rg --hidden "DB_PASSWORD"       # tìm trong hidden files (.env)

# ── fd — find thông minh ──
$ fd ".java"                       # tìm tất cả .java
$ fd -t d "controller"            # tìm thư mục tên controller
$ fd --changed-within 1d         # file thay đổi trong 1 ngày
$ fd -e yml -x bat              # mở tất cả .yml với bat

# ── fzf — fuzzy finder tương tác ──
$ fzf                             # chọn file từ danh sách
$ code $(fzf)                     # chọn file rồi mở VSCode

# fzf search trong history (Ctrl+R override)
$ history | fzf

# ── Tìm file lớn ──
$ find . -name "*.log" -size +100M   # file .log lớn hơn 100MB
$ find . -mtime -7 -name "*.java"   # java file sửa trong 7 ngày
$ find . -name "*.class" -delete     # tìm và xóa class files

HISTORY

History & fzf — Tái sử dụng lệnh cũ

Không gõ lại lệnh phức tạp — tìm và reuse trong giây lát

History tricks và fzf magic
History commands + fzf integrationHistory
# ── History cơ bản ──
$ history                       # xem tất cả history
$ history 50                    # 50 lệnh cuối
$ history | grep "docker"      # lọc history chứa "docker"
$ !!                             # chạy lại lệnh trước đó
$ !docker                       # chạy lại lệnh cuối bắt đầu bằng "docker"
$ !123                          # chạy lại lệnh số 123
$ sudo !!                       # chạy lại lệnh trước với sudo

# ── Config history tốt hơn trong ~/.zshrc ──
HISTSIZE=50000
SAVEHIST=50000
HISTFILE=~/.zsh_history
setopt HIST_IGNORE_DUPS         # không lưu duplicate
setopt HIST_IGNORE_SPACE        # lệnh bắt đầu bằng space không lưu (passwords!)
setopt SHARE_HISTORY           # chia sẻ history giữa các tab
setopt HIST_VERIFY             # hiện lệnh trước khi thực hiện !n

# ── fzf magic ──
# Ctrl+R → fuzzy search history (cực mạnh!)
# Ctrl+T → fuzzy search file trong thư mục hiện tại
# Alt+C  → fuzzy cd vào thư mục con

# Kill process bằng fzf
$ kill -9 $(ps aux | fzf | awk '{print $2}')

# Checkout git branch bằng fzf
$ git checkout $(git branch | fzf)

# Preview file khi tìm với fzf
$ fzf --preview 'bat --color=always {}'
ALIAS

Alias & Functions — Tạo lệnh riêng

Rút gọn lệnh dài thành alias ngắn và viết function tái sử dụng

Functions hữu ích cho Backend Developer
~/.zshrc — Alias và Functions thực tếFunctions
# ── Thêm vào ~/.zshrc ──

# mkcd — tạo thư mục rồi vào luôn
function mkcd() { mkdir -p $1 && cd $1 ; }
# Dùng: mkcd ~/Projects/new-microservice

# port — xem process nào đang dùng port
function port() { lsof -i :$1 ; }
# Dùng: port 8080

# killport — kill process đang chiếm port
function killport() {
    lsof -ti :$1 | xargs kill -9
    echo "✅ Killed process on port $1"
}
# Dùng: killport 8080

# jwt — decode JWT token (không cần internet)
function jwt() {
    echo $1 | cut -d. -f2 | base64 -d 2>/dev/null | jq .
}
# Dùng: jwt "eyJhbGci..."  → in payload JSON đẹp

# envload — load .env file
function envload() {
    export $(grep -v '^#' ${1:-.env} | xargs)
    echo "✅ Loaded ${1:-.env}"
}
# Dùng: envload .env.local

# gclone — clone và cd vào luôn
function gclone() {
    git clone $1 && cd $(basename $1 .git)
}
# Dùng: gclone https://github.com/company/service.git

# backup — tạo backup nhanh
function backup() {
    cp $1 "${1}.bak.$(date +%Y%m%d_%H%M%S)"
    echo "✅ Backed up $1"
}
# Dùng: backup application.yml

# springrun — chạy Spring Boot với profile
function springrun() {
    ./mvnw spring-boot:run -Dspring-boot.run.profiles=${1:-local}
}
# Dùng: springrun dev  /  springrun local

# dklogs — docker logs với grep
function dklogs() {
    docker logs -f --tail=200 $1 | grep -i "${2:-error}"
}
# Dùng: dklogs my-api ERROR   (chỉ in dòng có ERROR)
TMUX

Tmux — Terminal Multiplexer

Session persist khi SSH disconnect · Nhiều pane trên 1 màn hình

Workflow Tmux cho Backend Dev
Tmux — session, window, pane managementTmux
# ── Cài đặt ──
$ brew install tmux

# ── Session management ──
$ tmux                           # tạo session mới
$ tmux new -s backend            # tạo session tên "backend"
$ tmux ls                        # liệt kê sessions
$ tmux attach -t backend        # vào lại session "backend"
$ tmux kill-session -t backend  # xóa session

# ── Prefix key: Ctrl+B (mặc định), đổi sang Ctrl+A (thuận hơn) ──

# ── Window (Tab) ──
# Ctrl+B c    → tạo window mới
# Ctrl+B 1..9 → chuyển window
# Ctrl+B ,    → đặt tên window
# Ctrl+B w    → list windows
# Ctrl+B &    → đóng window hiện tại

# ── Pane (Split) ──
# Ctrl+B %    → chia dọc
# Ctrl+B "    → chia ngang
# Ctrl+B ←↑↓→ → chuyển pane
# Ctrl+B z    → zoom pane (phóng to/thu nhỏ)
# Ctrl+B x    → đóng pane
# Ctrl+B d    → detach (session vẫn chạy)

# ── ~/.tmux.conf — cấu hình tốt hơn ──
set -g prefix C-a               # đổi prefix thành Ctrl+A
unbind C-b
bind C-a send-prefix

set -g mouse on               # bật chuột (click để chọn pane)
set -g history-limit 10000
setw -g mode-keys vi          # vi-style copy mode

# Split bằng | và -
bind | split-window -h
bind - split-window -v

# ── Backend dev tmux session layout ──
$ tmux new -s myproject         # Window 1: code
# Ctrl+B c                       # Window 2: run spring
# Ctrl+B c                       # Window 3: docker
# Ctrl+B c                       # Window 4: logs

Docker CLI — container, image, network, volumeDocker
# ── Images ──
$ docker images                          # danh sách images
$ docker pull postgres:16               # pull image
$ docker build -t my-app:1.0 .          # build image
$ docker rmi my-app:1.0                 # xóa image
$ docker image prune -a                  # dọn images không dùng

# ── Containers ──
$ docker ps                               # containers đang chạy
$ docker ps -a                            # tất cả containers
$ docker run -d -p 5432:5432 \
    -e POSTGRES_PASSWORD=secret \
    --name postgres postgres:16
$ docker stop postgres                   # dừng container
$ docker start postgres                  # khởi động lại
$ docker rm -f postgres                 # force xóa
$ docker container prune                  # dọn containers stopped

# ── Exec vào container ──
$ docker exec -it postgres bash         # vào bash
$ docker exec -it postgres psql -U admin# vào psql trực tiếp
$ docker exec my-app env                # xem env vars

# ── Logs ──
$ docker logs my-app                     # xem logs
$ docker logs -f --tail=100 my-app      # follow 100 dòng cuối
$ docker logs my-app --since=30m        # logs 30 phút gần đây

# ── Docker Compose ──
$ docker compose up -d                  # start tất cả services (detached)
$ docker compose up -d postgres kafka   # chỉ start chỉ định
$ docker compose down                    # stop tất cả
$ docker compose down -v                # stop + xóa volumes
$ docker compose logs -f api            # follow logs service api
$ docker compose ps                      # status

#!/usr/bin/env bash
# deploy.sh — Deploy Spring Boot service lên server
set -euo pipefail  # e=exit on error, u=undefined vars, o=pipe fail

# ── Colors ──
GREEN='\033[0;32m'; RED='\033[0;31m'; NC='\033[0m'
info()  { echo -e "${GREEN}[INFO]${NC} $1"; }
error() { echo -e "${RED}[ERROR]${NC} $1"; exit 1; }

# ── Config ──
APP_NAME="order-service"
ENV=${1:-dev}              # default: dev
VERSION=$(git describe --tags --abbrev=0)

info "Deploying $APP_NAME v$VERSION to $ENV..."

# ── Build ──
./mvnw clean package -DskipTests -Pprod || error "Build failed"

# ── Build Docker image ──
docker build -t "$APP_NAME:$VERSION" . || error "Docker build failed"

# ── Push to registry ──
docker tag "$APP_NAME:$VERSION" "registry.company.com/$APP_NAME:$VERSION"
docker push "registry.company.com/$APP_NAME:$VERSION"

info "✅ Deploy $APP_NAME:$VERSION to $ENV completed!"

#!/usr/bin/env bash
# healthcheck.sh — Kiểm tra health của service
URL=${1:-http://localhost:8080/actuator/health}
MAX_RETRIES=30; WAIT=2; COUNT=0

echo "⏳ Waiting for service at $URL..."
until curl -sf "$URL" | jq -e '.status == "UP"' > /dev/null; do
    COUNT=$(( COUNT + 1 ))
    [[ $COUNT -ge $MAX_RETRIES ]] && { echo "❌ Service failed to start"; exit 1; }
    echo "  Retry $COUNT/$MAX_RETRIES..."
    sleep $WAIT
done
echo "✅ Service is UP!"

#!/usr/bin/env bash
# db-backup.sh — Backup PostgreSQL hàng ngày
DB_HOST="postgres.prod"
DB_NAME="shopdb"
BACKUP_DIR="$HOME/backups"
DATE=$(date +%Y%m%d_%H%M%S)
FILE="$BACKUP_DIR/${DB_NAME}_$DATE.sql.gz"

mkdir -p $BACKUP_DIR
pg_dump -h $DB_HOST -U admin $DB_NAME | gzip > $FILE
echo "✅ Backup saved: $FILE ($(du -sh $FILE | cut -f1))"

# Xóa backup cũ hơn 7 ngày
find $BACKUP_DIR -mtime +7 -name "*.sql.gz" -delete

# ── Tạo SSH key ──
$ ssh-keygen -t ed25519 -C "your@email.com"     # Ed25519 (hiện đại hơn RSA)

# ── ~/.ssh/config — cấu hình shortcut ──
Host dev
    HostName       server.dev.company.com
    User           ubuntu
    IdentityFile   ~/.ssh/id_dev
    ServerAliveInterval 60                         # prevent disconnect
    ServerAliveCountMax 3

Host prod-bastion
    HostName       bastion.prod.company.com
    User           ec2-user
    IdentityFile   ~/.ssh/id_prod

Host prod-api
    HostName       10.0.1.50                       # private IP
    User           ubuntu
    ProxyJump      prod-bastion                    # qua bastion
    IdentityFile   ~/.ssh/id_prod

# ── Kết nối ──
$ ssh dev                       # kết nối server dev (dùng config)
$ ssh prod-api                  # tự qua bastion!

# ── SCP — Copy files ──
$ scp deploy.jar dev:/opt/app/  # copy file lên server
$ scp dev:/var/log/app.log .   # copy file về local
$ scp -r dev:/opt/app/ backup/ # copy cả thư mục

# rsync — hiệu quả hơn scp (chỉ sync phần thay đổi)

1

Cài Homebrew + iTerm2

2

3

Cài các tools thay thế: eza, bat, rg, fd, fzf

4

Cấu hình ~/.zshrc với aliases và functions

5

Setup ~/.ssh/config cho tất cả servers

Cấu hình shortcut cho dev, staging, prod. Dùng ProxyJump cho server qua bastion. Thêm key vào Keychain để không gõ passphrase.

6

Học Tmux nếu làm việc nhiều với remote servers

Tmux giữ session khi SSH disconnect. Tạo layout riêng cho từng project: window code, window run, window logs, window docker.
